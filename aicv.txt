var app = angular.module("rApp", []);
app.controller("rCtrl", function ($scope) {
$scope.process = function () {
var n = $scope.num;
if (n === undefined || n === null) {
$scope.reversed = "";
$scope.sum = "";
return;
}
var s = Math.abs(parseInt(n)).toString();
$scope.reversed = s.split("").reverse().join("");
var total = 0;
for (var i = 0; i < s.length; i++) {
total += parseInt(s[i]);
}
$scope.sum = total;
};
});
